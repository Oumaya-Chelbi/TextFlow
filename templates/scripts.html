Objectif : Télécharger des articles scientifiques depuis arXiv (catégorie cs.CL)

import requests
import os
import feedparser

# Dossier où on va sauvegarder les PDFs
SAVE_DIR = "arxiv_pdfs"
os.makedirs(SAVE_DIR, exist_ok=True)  # Crée le dossier s'il n'existe pas

# Requête pour récupérer des articles en informatique (cs.CL = Computation and Language)
QUERY = "cat:cs.CL"
MAX_RESULTS = 650  # Nombre max d'articles qu'on veut

# Construction de l'URL pour interroger l'API arXiv
url = (
    "http://export.arxiv.org/api/query?"
    f"search_query={QUERY}&start=0&max_results={MAX_RESULTS}"
)

# On parse le résultat (format Atom/RSS)
feed = feedparser.parse(url)

# Pour chaque article trouvé
for entry in feed.entries:

    # On récupère l'ID de l'article (ex: 2401.12345)
    arxiv_id = entry.id.split("/")[-1]
    
    # Le deuxième lien est celui du PDF
    pdf_url = entry.links[1].href

    # Chemin où sauvegarder le fichier
    out_path = os.path.join(SAVE_DIR, arxiv_id + ".pdf")

    # Si le fichier existe déjà, on passe (évite de retélécharger)
    if os.path.exists(out_path):
        continue

    print("Téléchargement :", arxiv_id)

    # Téléchargement du PDF
    r = requests.get(pdf_url)
    
    # Sauvegarde en binaire (mode 'wb')
    with open(out_path, "wb") as f:
        f.write(r.content)
extract_arxiv_texts.py arXiv

Objectif : Convertir les PDF arXiv en fichiers texte exploitables

import os
import fitz   # PyMuPDF - bibliothèque pour lire les PDFs
import sys

# Dossiers d'entrée et de sortie
PDF_DIR = "arxiv_pdfs"
OUT_DIR = "arxiv_texts"

# Création du dossier de sortie
os.makedirs(OUT_DIR, exist_ok=True)


def extract_text(pdf_path):
    """Extrait le texte d'un PDF"""
    
    try:
        # On ouvre le PDF
        doc = fitz.open(pdf_path)
        
        # Vérification que le PDF est valide
        if doc.is_closed or doc.needs_pass:
            print(f"  ⚠ Document invalide ou protégé: {pdf_path}")
            return None
            
        full_text = []
        
        # On parcourt chaque page du PDF
        for page_num in range(len(doc)):
            try:
                page = doc[page_num]
                text = page.get_text()  # Extraction du texte de la page
                
                # On ne garde que les pages non vides
                if text and text.strip():
                    full_text.append(text)
                    
            except Exception as e:
                print(f"  ⚠ Erreur sur la page {page_num} de {os.path.basename(pdf_path)}: {str(e)}")
                continue
        
        # On ferme le document
        doc.close()
        
        # Si on n'a rien extrait, on retourne None
        if not full_text:
            print(f"  Aucun texte extrait de: {os.path.basename(pdf_path)}")
            return None
            
        # On retourne tout le texte en une seule chaîne
        return "\n".join(full_text)
        
    except Exception as e:
        print(f"   Erreur lors de l'extraction de {os.path.basename(pdf_path)}: {str(e)}")
        return None


def extract_text_alternative(pdf_path):
    """Méthode alternative si la première échoue"""
    
    try:
        doc = fitz.open(pdf_path)
        full_text = []
        
        for page_num in range(len(doc)):
            try:
                page = doc[page_num]
                
                # Première méthode : extraction en texte brut
                text = page.get_text("text")
                
                # Si ça ne marche pas, on essaye mot par mot
                if not text or not text.strip():
                    text = page.get_text("words")  # Extraction mot par mot
                    if text:
                        # On reconstruit le texte à partir des mots
                        text = " ".join([word[4] for word in text if len(word) > 4])
                
                if text and text.strip():
                    full_text.append(text)
                    
            except Exception as e:
                print(f" Erreur méthode alternative page {page_num}: {str(e)}")
                continue
                
        doc.close()
        
        if full_text:
            return "\n".join(full_text)
        return None
        
    except Exception as e:
        print(f" Erreur méthode alternative: {str(e)}")
        return None


def process_pdfs():
    """Traite tous les PDFs du répertoire"""
    
    # Compteurs pour les statistiques
    total_pdfs = 0
    success_count = 0
    failed_count = 0
    corrupted_count = 0
    
    # On liste tous les fichiers PDF
    pdf_files = [f for f in os.listdir(PDF_DIR) if f.endswith(".pdf")]
    total_pdfs = len(pdf_files)
    
    print(f"\n {total_pdfs} fichiers PDF trouvés dans {PDF_DIR}\n")
    
    # On traite chaque fichier un par un
    for i, file in enumerate(pdf_files, 1):
        pdf_path = os.path.join(PDF_DIR, file)
        
        print(f" [{i}/{total_pdfs}] Traitement: {file}")
        
        # Essai de la méthode standard
        txt = extract_text(pdf_path)
        
        # Si ça échoue, on essaye la méthode alternative
        if txt is None:
            print(f"   Tentative avec méthode alternative...")
            txt = extract_text_alternative(pdf_path)
            if txt is not None:
                print(f"  Réussi avec méthode alternative!")
        
        # Si on a réussi à extraire du texte
        if txt is not None and txt.strip():
            out_path = os.path.join(OUT_DIR, file.replace(".pdf", ".txt"))
            
            try:
                # Sauvegarde du texte dans un fichier .txt
                with open(out_path, "w", encoding="utf8") as f:
                    f.write(txt)
                    
                print(f" Sauvegardé: {os.path.basename(out_path)} ({len(txt)} caractères)")
                success_count += 1
                
            except Exception as e:
                print(f"  Erreur sauvegarde: {str(e)}")
                failed_count += 1
        else:
            # Si l'extraction a échoué
            try:
                file_size = os.path.getsize(pdf_path)
                if file_size < 1000:
                    print(f" Fichier suspect (taille: {file_size} octets)")
                else:
                    print(f" Échec extraction")
                corrupted_count += 1
            except:
                corrupted_count += 1
    
    # Rapport final
    print(f"\n" + "="*50)
    print(f" RAPPORT FINAL")
    print(f"="*50)
    print(f"Total PDFs trouvés: {total_pdfs}")
    print(f"Extractions réussies: {success_count}")
    print(f"Échecs extraction: {failed_count}")
    print(f"Fichiers corrompus: {corrupted_count}")
    print(f"Textes sauvegardés dans: {OUT_DIR}")
    print(f"="*50)


if __name__ == "__main__":
    
    # Vérification que le dossier source existe
    if not os.path.exists(PDF_DIR):
        print(f" Erreur: Le répertoire {PDF_DIR} n'existe pas!")
        sys.exit(1)
    
    print(" Début du traitement des PDFs arXiv")
    print(f" Source: {PDF_DIR}")
    print(f" Destination: {OUT_DIR}")
    
    # On lance le traitement
    process_pdfs()
    
    print("\n Traitement terminé!")
download_pmc_xml.py PMC

Objectif : Télécharger des articles biomédicaux depuis PubMed Central

import requests
import os
import time

# Dossier où on va sauvegarder les fichiers XML
SAVE_DIR = "pmc_xml"
os.makedirs(SAVE_DIR, exist_ok=True)

# ==========================================
# ÉTAPE 1 : Recherche des IDs des articles
# ==========================================

# URL de recherche des articles PMC
search_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi"

# Paramètres de la requête
params = {
    "db": "pmc",                    # Base de données : PubMed Central
    "term": "open access[filter]",  # On veut uniquement les articles en libre accès
    "retmax": 650                   # Nombre max d'articles
}

print("Recherche des IDs PMC...")

# On envoie la requête
r = requests.get(search_url, params=params, timeout=30)
r.raise_for_status()  # Lève une erreur si la requête a échoué

# On extrait les IDs de la réponse XML
ids = []
for line in r.text.split("")[1:]:
    ids.append(line.split("")[0])

print(len(ids), "IDs trouvés")

# ==========================================
# ÉTAPE 2 : Téléchargement des fichiers XML
# ==========================================

# Pour chaque ID trouvé
for pmc_id in ids:

    # URL pour récupérer le contenu de l'article
    fetch_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/efetch.fcgi"

    params = {
        "db": "pmc",
        "id": pmc_id,
        "retmode": "xml"   # On veut le format XML
    }

    out_path = os.path.join(SAVE_DIR, pmc_id + ".xml")

    # Si le fichier existe déjà, on passe (évite de retélécharger)
    if os.path.exists(out_path):
        print("Déjà présent :", pmc_id)
        continue

    print("Téléchargement PMC", pmc_id)

    try:
        # Téléchargement du fichier XML
        r = requests.get(fetch_url, params=params, timeout=30)

        if r.status_code != 200:
            print("  -> erreur HTTP", r.status_code)
            continue

        # Sauvegarde du fichier
        with open(out_path, "w", encoding="utf-8") as f:
            f.write(r.text)

    except requests.exceptions.RequestException as e:
        print("  -> erreur réseau pour", pmc_id)
        print("     ", e)
        continue

    # Pause pour éviter de surcharger le serveur
    time.sleep(1)

generate_wiki.py Wikipédia

Objectif : Générer des exemples d'insertion à partir des articles Wikipédia

import os
import json
import random
from tqdm import tqdm

# Configuration
INPUT_DIR = "extracted"        # Dossier où sont les articles Wikipédia extraits
OUTPUT_FILE = "wiki_dataset.jsonl"  # Fichier de sortie

MIN_PARAGRAPHS = 5             # Minimum de paragraphes pour garder un article
MAX_SAMPLES_PER_ARTICLE = 1    # Nombre d'exemples par article


def clean_paragraphs(text):
    """Nettoie le texte en gardant uniquement les paragraphes valides"""
    
    # On découpe par sauts de ligne
    paragraphs = text.split("\n")
    
    # On ne garde que les paragraphes non vides et assez longs
    paragraphs = [
        p.strip()
        for p in paragraphs
        if len(p.strip()) > 80  # On ignore les paragraphes trop courts
    ]
    
    return paragraphs


def generate_examples_from_text(title, text):
    """
    À partir d'un article, génère des exemples où on retire un paragraphe
    et on demande au modèle de le réinsérer
    """
    
    # On nettoie le texte
    paragraphs = clean_paragraphs(text)
    
    # Si l'article n'a pas assez de paragraphes, on l'ignore
    if len(paragraphs) < MIN_PARAGRAPHS:
        return []
    
    examples = []
    
    # On génère un exemple par article (ou plusieurs selon config)
    for _ in range(min(MAX_SAMPLES_PER_ARTICLE, len(paragraphs))):
        
        # On choisit un paragraphe au hasard à retirer
        idx = random.randint(0, len(paragraphs) - 1)
        
        # Le paragraphe qu'on va retirer
        removed = paragraphs[idx]
        
        # L'article sans ce paragraphe
        truncated = paragraphs[:idx] + paragraphs[idx + 1:]
        
        # On crée l'exemple
        example = {
            "source": "wikipedia",               # Source des données
            "title": title,                      # Titre de l'article
            "article_truncated": truncated,      # Article sans le paragraphe
            "paragraph_to_insert": removed,      # Paragraphe à insérer
            "true_position": idx                 # Position correcte (0-based)
        }
        
        examples.append(example)
    
    return examples


def process_wiki_dump():
    """Traitement principal"""
    
    # On récupère tous les fichiers d'articles Wikipédia
    all_files = []
    
    for root, _, files in os.walk(INPUT_DIR):
        for file in files:
            all_files.append(os.path.join(root, file))
    
    print(len(all_files), "fichiers trouvés")
    
    # On mélange pour avoir des articles variés
    random.shuffle(all_files)
    
    article_count = 0  # Compteur d'articles traités
    
    with open(OUTPUT_FILE, "w", encoding="utf-8") as out:
        
        # On parcourt tous les fichiers
        for file_path in tqdm(all_files):
            
            # On s'arrête quand on a assez d'articles
            if article_count >= MAX_ARTICLES:
                break
            
            # On ouvre le fichier
            with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
                
                # Chaque ligne est un article au format JSON
                for line in f:
                    
                    if article_count >= MAX_ARTICLES:
                        break
                    
                    line = line.strip()
                    if not line:
                        continue
                    
                    try:
                        article = json.loads(line)
                    except json.JSONDecodeError:
                        continue
                    
                    # On génère les exemples pour cet article
                    examples = generate_examples_from_text(
                        article.get("title", ""),
                        article.get("text", "")
                    )
                    
                    # Si l'article a donné des exemples, on le compte
                    if examples:
                        article_count += 1
                    
                    # On écrit les exemples dans le fichier de sortie
                    for ex in examples:
                        out.write(json.dumps(ex, ensure_ascii=False) + "\n")
    
    print("Articles utilisés :", article_count)


if __name__ == "__main__":
    process_wiki_dump()

    """
    Extrait les paragraphes d'un fichier XML PMC

    """

    tree = ET.parse(xml_path)

    return paragraphs

import json
import random

# Fichier d'entrée
INPUT_FILE = "dataset_shuffled.jsonl"

# Fichiers de sortie
TRAIN_FILE = "train.jsonl"
DEV_FILE = "dev.jsonl"
TEST_FILE = "test.jsonl"

# Lecture du dataset
with open(INPUT_FILE, encoding="utf8") as f:
    data = [json.loads(line) for line in f]

# On mélange à nouveau pour garantir un split aléatoire
random.shuffle(data)

# Répartition : 80% train, 10% dev, 10% test

dev = data[int(0.8 * n): int(0.9 * n)]
test = data[int(0.9 * n):]

# Sauvegarde des trois fichiers
for name, dataset in [
    (TRAIN_FILE, train),
    (DEV_FILE, dev),
    (TEST_FILE, test)
]:
    with open(name, "w", encoding="utf8") as f:
        for ex in dataset:
            f.write(json.dumps(ex, ensure_ascii=False) + "\n")

print("Train:", len(train))
print("Dev:", len(dev))
print("Test:", len(test))
build_pairs.py Traitement

Objectif : Transformer les articles en paires (prev, insert, next) pour l'entraînement

import json
import argparse

def convert_dataset(input_file, output_file):
    """
    Transforme un dataset d'articles en paires d'insertion
    
    Pour chaque article, on génère des paires (prev, next, insert, label)
    où :
    - prev = paragraphe avant l'insertion
    - next = paragraphe après l'insertion
    - insert = paragraphe à insérer
    - label = 1 si c'est la bonne position, 0 sinon
    """
    
    # On ouvre le fichier d'entrée (dataset brut)
    # et le fichier de sortie où on va écrire les paires
    with open(input_file, encoding="utf8") as f, open(output_file, "w", encoding="utf8") as out:
        
        # On lit le fichier ligne par ligne (format JSONL)
        for line in f:
            
            # On convertit la ligne JSON en dictionnaire Python
            ex = json.loads(line)
            
            # On récupère les éléments utiles
            article = ex["article_truncated"]       # liste des paragraphes
            insert = ex["paragraph_to_insert"]      # paragraphe à insérer
            gold = ex["true_position"]              # position correcte
            
            # Si l'article est trop court, on ne peut pas créer de paires
            if len(article) < 2:
                continue
            
            # On parcourt toutes les positions possibles dans l'article
            for i in range(len(article) - 1):
                
                # On récupère le paragraphe avant et après
                prev_p = article[i]
                next_p = article[i + 1]
                
                # Label = 1 si c'est la bonne position
                # Attention : gold correspond à la position dans l'article original
                # On compare avec i+1 car les positions sont décalées
                label = 1 if i == gold - 1 else 0
                
                # On construit la paire pour le modèle
                pair = {
                    "prev": prev_p,
                    "next": next_p,
                    "insert": insert,
                    "label": label
                }
                
                # On écrit la paire dans le fichier de sortie (format JSONL)
                out.write(json.dumps(pair, ensure_ascii=False) + "\n")

Rétrécissement du dataset

Face aux contraintes de mémoire GPU et pour accélérer les phases d'expérimentation, nous avons dû réduire la taille de notre dataset tout en préservant son équilibre entre les classes. Cette étape, appelée sous-échantillonnage (downsampling), nous permet de travailler avec un jeu de données plus petit mais toujours représentatif.

Nous avons utilisé la méthode du réservoir sampling (échantillonnage par réservoir) qui garantit un tirage aléatoire uniforme même sur des fichiers trop volumineux pour être chargés entièrement en mémoire.

L'objectif était de créer des versions réduites de nos datasets :

  • Train : 20 000 exemples par classe (40 000 au total)
  • Dev (validation) : 2 000 exemples par classe (4 000 au total)
  • Test : 5 000 exemples par classe (10 000 au total)

Cette réduction nous a permis de :

  • Réduire drastiquement le temps d'entraînement (de plusieurs heures à 1 seule heure)
  • Adapter la consommation mémoire aux limitations de nos GPUs
reservoir_sampling.py Traitement

Objectif : Réduire la taille du dataset tout en préservant l'équilibre entre les classes (label 0 et label 1).

import json
import random
from pathlib import Path

def reservoir_sample_by_label(input_path, output_path, target_counts, seed=42):
    """
    Échantillonne un dataset JSONL en préservant un nombre cible d'exemples par label.
    
    Args:
        input_path: chemin du fichier d'entrée
        output_path: chemin du fichier de sortie
        target_counts: dictionnaire {label: nombre_cible}
        seed: graine aléatoire pour la reproductibilité
    """
    
    # On fixe la graine aléatoire pour la reproductibilité
    random.seed(seed)

    # Création des réservoirs pour chaque label
    reservoirs = {label: [] for label in target_counts}
    seen_counts = {label: 0 for label in target_counts}

    print(f"Lecture du fichier: {input_path}")
    
    with open(input_path, "r", encoding="utf-8") as f:
        for line in f:
            obj = json.loads(line)
            label = obj["label"]

            # On ignore les labels non ciblés
            if label not in target_counts:
                continue

            seen_counts[label] += 1
            k = target_counts[label]

            # Algorithme du réservoir sampling
            if len(reservoirs[label]) < k:
                # Réservoir pas encore plein → on ajoute directement
                reservoirs[label].append(line)
            else:
                # Réservoir plein → remplacement aléatoire avec probabilité k/n
                j = random.randint(1, seen_counts[label])
                if j <= k:
                    reservoirs[label][j - 1] = line

    # Vérification que tous les réservoirs sont remplis
    for label, target in target_counts.items():
        actual = len(reservoirs[label])
        if actual < target:
            print(f" Attention: label {label} n'a que {actual} exemples (objectif: {target})")

    # Mélange des exemples de tous les labels
    all_lines = []
    for label in target_counts:
        all_lines.extend(reservoirs[label])
    random.shuffle(all_lines)

    # Création du dossier de sortie si nécessaire
    Path(output_path).parent.mkdir(parents=True, exist_ok=True)
    
    # Écriture du fichier de sortie
    with open(output_path, "w", encoding="utf-8") as f:
        for line in all_lines:
            f.write(line)

    print(f" Dataset sauvegardé: {output_path}")
    print("   Répartition finale :")
    for label in target_counts:
        print(f"     label {label}: {len(reservoirs[label])} exemples")


if __name__ == "__main__":
    # Chemins des fichiers originaux
    TRAIN_IN = "dataset/train_pairs.jsonl"

Pourquoi l'échantillonnage par réservoir ?
Cette méthode présente plusieurs avantages :
- Économie de mémoire : on ne charge jamais tout le dataset en mémoire
- Échantillon aléatoire uniforme : chaque ligne a la même probabilité d'être sélectionnée
- Reproductibilité : la graine aléatoire (seed=42) garantit des résultats identiques à chaque exécution
- Équilibre préservé : on conserve exactement le nombre d'exemples souhaité par classe

XLM-RoBERTa Entraînement

Objectif : Entraîner le modèle XLM-RoBERTa (modèle multilingue performant) sur notre tâche d'insertion de paragraphe.

XLM-RoBERTa est un modèle multilingue pré-entraîné sur 100 langues, ce qui le rend particulièrement adapté pour traiter des textes en français comme en anglais. Contrairement à l'approche de ranking utilisée pour BERT Multilingual, ici nous abordons le problème comme une classification binaire classique : pour chaque position possible, le modèle doit prédire si c'est la bonne position (label 1) ou non (label 0).

Le script commence par charger les données au format JSONL, où chaque exemple contient les champs prev, insert, next et label. La fonction preprocess() prépare chaque exemple en concaténant les trois paragraphes sous la forme insert [SEP] prev [SEP] next (l'ordre est différent de l'approche ranking pour mieux correspondre à l'architecture du modèle). La tokenisation utilise longest_first pour gérer les séquences de longueur variable.

Le modèle XLM-RoBERTa est configuré avec num_labels=2 pour effectuer une classification binaire. L'entraînement utilise le Trainer de Hugging Face, qui gère automatiquement la boucle d'entraînement, l'évaluation et la sauvegarde des checkpoints. Les métriques calculées sont l'accuracy, le F1-score, la précision et le rappel.

Les hyperparamètres sont les suivants : batch size de 16, learning rate de 2e-5, 4 epochs, warmup ratio de 0.1, et une évaluation à chaque epoch. Le modèle sauvegarde automatiquement le meilleur checkpoint en fonction de la loss de validation.

À la fin de l'entraînement, le script évalue le modèle sur les datasets de validation et de test, puis sauvegarde le modèle final dans my_xlm_roberta_model. Une fonction predict() est également fournie pour faire des prédictions sur de nouveaux exemples.

def preprocess(example):

    Tokenise un exemple pour XLM-RoBERTa

        text=text_primary,
        text_pair=text_secondary,
        truncation="longest_first",
        padding="max_length",
        max_length=256

def predict(prev, insert, next_text):

    Prédit la probabilité que la position soit correcte
    Retourne un score entre 0 et 1
    \"\"\"
    text_primary = str(insert)
    text_secondary = str(prev) + f" {tokenizer.sep_token} " + str(next_text)

        text=text_primary,
        text_pair=text_secondary,
        return_tensors="pt",
        truncation="longest_first",
        max_length=256

    inputs = {k: v.to(model.device) for k, v in inputs.items()}

        outputs = model(**inputs)

    probs = torch.softmax(outputs.logits, dim=1)
    return probs[0][1].item()  # Probabilité de la classe 1 (bonne position)

BertMultilingual Entraînement

Objectif : Entraîner le modèle BERT Multilingual sur la tâche de ranking

Le script train_ranking.py met en place un entraînement basé sur une approche de ranking pour apprendre à insérer correctement un paragraphe dans un article. Il commence par charger un fichier de données au format JSONL contenant des exemples déjà construits, où chaque ligne correspond à un triplet de paragraphes (prev, insert, next) ainsi qu’un label indiquant si la position est correcte ou non. Ces exemples sont ensuite séparés en deux groupes : les exemples positifs (où le paragraphe est bien placé) et les exemples négatifs (où il ne l’est pas). L’idée est de créer des paires composées d’un exemple positif et d’un exemple négatif, afin que le modèle apprenne à comparer les deux.

Pour chaque paire, le script prépare les entrées du modèle en concaténant les trois paragraphes sous la forme prev [SEP] insert [SEP] next, en utilisant le tokenizer associé au modèle bert-base-multilingual-cased. Cette étape transforme le texte en séquences de tokens numériques que BERT peut traiter. Le modèle est ensuite configuré pour produire un score unique (et non une classe), qui représente à quel point la position proposée est pertinente. Ainsi, chaque exemple (positif ou négatif) reçoit un score.

Pendant l’entraînement, pour chaque paire (positif, négatif), le script calcule une margin ranking loss. Le principe est simple : on veut que le score de l’exemple positif soit supérieur à celui de l’exemple négatif d’au moins une certaine marge. Si ce n’est pas le cas, une pénalité est appliquée, ce qui pousse le modèle à corriger ses prédictions. Ce mécanisme force le modèle à apprendre à distinguer les bonnes positions des mauvaises en les comparant directement, plutôt que de prendre des décisions isolées.

Enfin, à la fin de chaque epoch, une phase d’évaluation est réalisée. Pour chaque paire, on vérifie si le score attribué à l’exemple positif est bien supérieur à celui de l’exemple négatif. Si c’est le cas, la prédiction est considérée comme correcte. On calcule alors une accuracy de ranking, qui mesure la proportion de comparaisons correctement effectuées. Cette métrique est plus adaptée que l’accuracy classique, car elle reflète directement la capacité du modèle à choisir la meilleure position parmi plusieurs possibilités. Ainsi, tout le script est organisé autour de cette idée centrale : apprendre à classer des positions entre elles plutôt qu’à les juger indépendamment. 

# train_ranking.py
import json
import torch
import random
from torch.utils.data import Dataset, DataLoader
from transformers import AutoTokenizer, AutoModelForSequenceClassification
from torch.optim import AdamW
from tqdm import tqdm
import os

MODEL_NAME = "bert-base-multilingual-cased"  # Modèle multilingue
TRAIN_FILE = "dataset_prêt_utilisation/train_pairs.jsonl"
DEV_FILE = "dataset_prêt_utilisation/dev_pairs.jsonl"

BATCH_SIZE = 16      # Nombre d'exemples par batch
EPOCHS = 3           # Nombre de passages sur le dataset
LR = 2e-5            # Learning rate (vitesse d'apprentissage)
MAX_LEN = 512        # Longueur max des séquences

SAVE_PATH = "model_ranking"  # Dossier où sauvegarder le modèle


class RankingDataset(Dataset):
    """
    Dataset pour l'apprentissage par ranking
    Chaque exemple est une paire (exemple positif, exemple négatif)
    """
    
    def __init__(self, file, tokenizer):
        self.tokenizer = tokenizer
        self.pos_examples = []  # Exemples où label = 1 (bonne position)
        self.neg_examples = []  # Exemples où label = 0 (mauvaise position)

        print(f"Chargement: {file}")

        # On charge le fichier
        with open(file, "r", encoding="utf-8") as f:
            for line in f:
                ex = json.loads(line)
                if ex["label"] == 1:
                    self.pos_examples.append(ex)
                else:
                    self.neg_examples.append(ex)

        print(f"Positifs: {len(self.pos_examples)} | Négatifs: {len(self.neg_examples)}")

    def __len__(self):
        # Le dataset a la taille du nombre d'exemples positifs
        return len(self.pos_examples)

    def encode(self, ex):
        """Tokenise un exemple pour le modèle"""
        
        # On utilise [SEP] comme séparateur (token standard de BERT)
        sep = self.tokenizer.sep_token or "[SEP]"
        
        # On reconstruit le texte comme pendant l'entraînement
        text = f"{ex['prev']} {sep} {ex['insert']} {sep} {ex['next']}"
        
        # Tokenisation
        return self.tokenizer(
            text,
            truncation=True,
            padding="max_length",
            max_length=MAX_LEN,
            return_tensors="pt"
        )

    def __getitem__(self, idx):
        """
        Retourne un tuple (exemple positif, exemple négatif aléatoire)
        """
        # On prend un exemple positif
        pos = self.pos_examples[idx]
        
        # On choisit un exemple négatif au hasard
        neg = random.choice(self.neg_examples)
        
        # On encode les deux
        pos_enc = self.encode(pos)
        neg_enc = self.encode(neg)
        
        # On enlève la dimension batch (squeeze)
        pos_item = {k: v.squeeze() for k, v in pos_enc.items()}
        neg_item = {k: v.squeeze() for k, v in neg_enc.items()}
        
        return pos_item, neg_item


def collate_fn(batch):
    """
    Fonction qui regroupe les exemples en batch
    """
    pos_batch = {}
    neg_batch = {}
    
    # On empile les tensors pour le batch
    for key in batch[0][0].keys():
        pos_batch[key] = torch.stack([item[0][key] for item in batch])
        neg_batch[key] = torch.stack([item[1][key] for item in batch])
    
    return pos_batch, neg_batch


def evaluate(model, dataset, tokenizer, device, num_samples=200):
    """
    Évalue le modèle sur un échantillon du dataset
    """
    model.eval()
    correct = 0
    
    # On teste sur num_samples exemples
    for i in range(min(num_samples, len(dataset.pos_examples))):
        
        # On prend un exemple positif et un négatif aléatoire
        pos = dataset.pos_examples[i]
        neg = random.choice(dataset.neg_examples)
        
        # Fonction locale pour encoder
        def encode(ex):
            sep = tokenizer.sep_token or "[SEP]"
            text = f"{ex['prev']} {sep} {ex['insert']} {sep} {ex['next']}"
            return tokenizer(text, return_tensors="pt", truncation=True, 
                           padding=True, max_length=MAX_LEN)
        
        pos_enc = encode(pos)
        neg_enc = encode(neg)
        
        # Envoi sur le device (GPU/CPU)
        pos_enc = {k: v.to(device) for k, v in pos_enc.items()}
        neg_enc = {k: v.to(device) for k, v in neg_enc.items()}
        
        # Prédiction
        with torch.no_grad():
            pos_score = model(**pos_enc).logits.squeeze()
            neg_score = model(**neg_enc).logits.squeeze()
        
        # Si le score du positif est plus élevé, c'est une bonne prédiction
        if pos_score > neg_score:
            correct += 1
    
    acc = correct / num_samples
    print(f"Ranking accuracy: {acc:.4f}")
    return acc


def train():
    """
    Fonction principale d'entraînement
    """
    
    # Détection du GPU si disponible
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    print(f"Device: {device}")
    
    # Chargement du tokenizer
    tokenizer = AutoTokenizer.from_pretrained(MODEL_NAME)
    
    # Chargement du modèle avec num_labels=1 (sortie = un score)
    model = AutoModelForSequenceClassification.from_pretrained(
        MODEL_NAME,
        num_labels=1
    ).to(device)
    
    # Chargement des datasets
    train_dataset = RankingDataset(TRAIN_FILE, tokenizer)
    dev_dataset = RankingDataset(DEV_FILE, tokenizer)
    
    # DataLoader pour le train
    train_loader = DataLoader(
        train_dataset,
        batch_size=BATCH_SIZE,
        shuffle=True,
        collate_fn=collate_fn
    )
    
    # Optimiseur AdamW
    optimizer = AdamW(model.parameters(), lr=LR)
    
    best_acc = 0
    os.makedirs(SAVE_PATH, exist_ok=True)
    
    # Boucle d'entraînement
    for epoch in range(EPOCHS):
        print(f"\nEpoch {epoch+1}")
        model.train()
        
        loop = tqdm(train_loader)
        
        for pos_batch, neg_batch in loop:
            # Envoi sur le device
            pos_batch = {k: v.to(device) for k, v in pos_batch.items()}
            neg_batch = {k: v.to(device) for k, v in neg_batch.items()}
            
            # Scores des exemples positifs et négatifs
            pos_score = model(**pos_batch).logits.squeeze()
            neg_score = model(**neg_batch).logits.squeeze()
            
            # Loss = margin ranking loss
            # On veut que pos_score > neg_score d'au moins margin
            margin = 1.0
            loss = torch.clamp(margin - pos_score + neg_score, min=0).mean()
            
            # Backpropagation
            loss.backward()
            optimizer.step()
            optimizer.zero_grad()
            
            # Affichage de la loss
            loop.set_postfix(loss=loss.item())

    model.eval()

    correct = 0
    total = 0

    with torch.no_grad():

        for batch in dataloader:
